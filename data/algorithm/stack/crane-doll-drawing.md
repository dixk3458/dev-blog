function solution(board, moves) {
  let answer = 0;
  const stack = [];

  for (let m of moves) {
    for (let i = 0; i < board.length; i++) {
      if (board[i][m - 1] !== 0) {
        if (stack.length > 0 && board[i][m - 1] === stack[stack.length - 1]) {
          answer = answer + 2;
          stack.pop();
        } else {
          stack.push(board[i][m - 1]);
        }

        board[i][m - 1] = 0;

        break;
      }
    }
  }
  return answer;
}
